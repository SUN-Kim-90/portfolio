int OrderCollection::deleteOrder(string name, string menu, int time) {
  for (auto iter = order_list->begin(); iter != order_list->end(); iter++) {
    if ((*iter)->getRestaurantName() == name &&
        (*iter)->getMenuName() == menu) {
      if ((time - (*iter)->getTime()) >= 6480) { // 108분
        order_list->erase(iter);
        return 1; // 배송완료
      }
      return 2; // 배송중
    }
  }
  return 0;
}
